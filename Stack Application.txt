#include <stdio.h>
#include <ctype.h>
#include <string.h>
#define MAX 100

char stack[MAX];
int top = -1;

void push(char c) { stack[++top] = c; }
char pop() { return stack[top--]; }
int precedence(char op) {
    switch(op) {
        case '+': case '-': return 1;
        case '*': case '/': return 2;
        case '^': return 3;
    }
    return 0;
}

void infixToPostfix(char* infix, char* postfix) {
    int i = 0, j = 0;
    char c;
    
    while((c = infix[i++]) != '\0') {
        if(isalnum(c)) postfix[j++] = c;
        else if(c == '(') push(c);
        else if(c == ')') {
            while(top != -1 && stack[top] != '(')
                postfix[j++] = pop();
            pop();
        } else {
            while(top != -1 && precedence(stack[top]) >= precedence(c))
                postfix[j++] = pop();
            push(c);
        }
    }
    while(top != -1) postfix[j++] = pop();
    postfix[j] = '\0';
}

int main() {
    char infix[MAX] = "a+b*(c^d-e)^(f+g*h)-i";
    char postfix[MAX];
    
    infixToPostfix(infix, postfix);
    printf("Infix: %s\n", infix);
    printf("Postfix: %s\n", postfix);
    return 0;
}
